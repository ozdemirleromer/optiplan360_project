type SidebarRole = "ADMIN" | "OPERATOR" | "STATION";
 
function toSidebarRole(role?: string): SidebarRole {
  const normalized = role?.toUpperCase().trim();
  if (!normalized) return "OPERATOR";
  if (normalized === "ADMIN" || normalized.includes("ADMIN")) return "ADMIN";
  if (normalized === "OPERATOR" || normalized.includes("OPERATOR")) return "OPERATOR";
  if (normalized === "STATION" || normalized === "KIOSK" || normalized.includes("KIOSK")) return "STATION";
  return "OPERATOR";
}

// Oturum acik olmayan kullanicilar buraya ulasamaz (App guard).
function AuthenticatedApp({ authUser }: { authUser: User }) {
  const [page, setPage] = useState<Page>("dashboard");
  const [sidebarCollapsed, setSidebarCollapsed] = useState(false);
  const [mobileMenuOpen, setMobileMenuOpen] = useState(false);
  const [editingOrder, setEditingOrder] = useState<Order | null>(null);
  const [spotlightOpen, setSpotlightOpen] = useState(false);
 
  // Global keyboard shortcuts
  useKeyboardShortcuts([
    { keys: ['Ctrl', 'k'], description: 'Arama aç', action: () => setSpotlightOpen(true) },
    { keys: ['Escape'], description: 'Kapat', action: () => setSpotlightOpen(false), preventDefault: false },
  ]);
 
  const toggleSidebar = () => {
    if (window.innerWidth <= 768) {
      setMobileMenuOpen(!mobileMenuOpen);
    } else {
      setSidebarCollapsed(!sidebarCollapsed);
    }
  };
 
  const logout = useAuthStore((s) => s.logout);
 
  const fetchOrders = useOrdersStore((s) => s.fetchOrders);
  const initialized = useOrdersStore((s) => s.initialized);
  const loading = useOrdersStore((s) => s.isLoading);
  const error = useOrdersStore((s) => s.error);
  const allOrders = useOrdersStore((s) => s.orders);
 
  const badgeCounts = useMemo(() => ({
    orders: allOrders.filter((o) => o.status === "NEW").length || undefined,
    kanban: allOrders.filter((o) => o.status === "IN_PRODUCTION").length || undefined,
    payment: undefined,
  }), [allOrders]);
 
  useEffect(() => {
    void fetchOrders();
  }, [fetchOrders]);
 
  const currentUser = useMemo(
    () => ({
      name: authUser.fullName || authUser.username || authUser.email || "Operator",
      role: toSidebarRole(authUser.role),
    }),
    [authUser]
  );
 
  const openEditor = (order: Order | null) => {
    setEditingOrder(order);
    setPage("order-editor");
  };
 
  const closeEditor = () => {
    setEditingOrder(null);
    setPage("orders");
  };
 
  const statusNote = !initialized || loading
    ? (
      <div style={{ display: "flex", alignItems: "center", gap: "8px" }}>
        <div className="loading-spinner" />
        Veriler yükleniyor...
      </div>
    )
    : error
      ? `Veri bağlantısı uyarısı: ${error}`
      : "";
 
  const renderPage = () => {
    switch (page) {
      case "dashboard":
        return <Dashboard currentUser={{ name: currentUser.name }} onNewOrder={() => openEditor(null)} />;
      case "orders":
        return <Orders onEdit={openEditor} />;
      case "order-editor":
        return <OrderEditor order={editingOrder} onBack={closeEditor} />;
      case "kanban":
        return <Kanban onEdit={(order) => openEditor(order)} />;
      case "ai-center":
        return <AICenterPage />;
      case "card-management":
        return <CardManagementPage />;
      case "reports-analytics":
        return <ReportsAnalyticsPage />;
      case "payment":
        return <PaymentDashboard />;
      case "integration-settings":
        return <IntegrationSettings />;
      case "integration-health":
        return <IntegrationHealth />;
      case "integrations":
        return <ModularIntegrationsPage />;
      case "stations":
        return <StationsPage />;
      case "system-logs":
        return <SystemLogsPage />;
      case "user-management":
        return <UserManagementPage />;
      case "config":
        return <ConfigPage />;
      case "organization":
        return <OrganizationPage />;
      case "user-activity":
        return <UserActivityPage />;
      case "workflows":
        return <WorkflowBuilder />;
      case "api-portal":
        return <ApiPortal />;
      default:
        return <InfoPage title="Sayfa Bulunamadı" detail="İstenen sayfa tanımlı değil." />;
    }
  };
 
  return (
    <div
      style={{
        minHeight: "100vh",
        background: COLORS.bg.main,
        color: COLORS.text,
        display: "flex",
        fontFamily: TYPOGRAPHY.fontFamily.base,
      }}
    >
      {/* Mobile Header - Only visible on mobile */}
      <div className="show-mobile" style={{ position: "fixed", top: 0, left: 0, right: 0, zIndex: 101 }}>
        <MobileHeader
          title="Optiplan360"
          subtitle={page.charAt(0).toUpperCase() + page.slice(1)}
          isMenuOpen={mobileMenuOpen}
          onMenuToggle={toggleSidebar}
          userName={currentUser.name}
          userRole={currentUser.role}
        />
      </div>
 
      {/* Mobile Overlay */}
      <div
        className={`sidebar-overlay ${mobileMenuOpen ? "is-visible" : ""}`}
        onClick={() => setMobileMenuOpen(false)}
      />
 
      <Sidebar
        page={page}
        onNav={(nextPage) => {
          setPage(nextPage as Page);
          setMobileMenuOpen(false);
        }}
        collapsed={sidebarCollapsed}
        onToggle={toggleSidebar}
        userRole={currentUser.role}
        currentUser={authUser}
        badgeCounts={badgeCounts}
        onLogout={() => {
          logout();
          setPage("dashboard");
          setEditingOrder(null);
        }}
      />
      <main className="page-content" style={{ flex: 1, minWidth: 0, marginTop: "var(--mobile-header-height, 0)" }}>
        {statusNote ? (
          <div
            style={{
              margin: "16px 20px 0",
              padding: "12px 16px",
              borderRadius: RADIUS.lg,
              border: `1px solid ${error ? COLORS.warning.DEFAULT : COLORS.info[500]}66`,
              background: error
                ? "linear-gradient(135deg, rgba(251,191,36,0.15) 0%, rgba(245,158,11,0.08) 100%)"
                : `linear-gradient(135deg, ${primaryRgba(0.15)} 0%, ${primaryRgba(0.08)} 100%)`,
              color: error ? COLORS.warning[800] : COLORS.info[800],
              fontSize: 13,
              fontWeight: 500,
              boxShadow: `0 2px 8px ${error ? "rgba(251,191,36,0.2)" : primaryRgba(0.2)}`,
              backdropFilter: "blur(8px)",
              display: "flex",
              alignItems: "center",
              gap: "8px",
            }}
          >
            <div
              style={{
                width: "8px",
                height: "8px",
                borderRadius: "50%",
                background: error ? COLORS.warning.DEFAULT : COLORS.info[500],
                animation: "pulse 2s infinite",
              }}
            />
            {statusNote}
          </div>
        ) : null}
        <ErrorBoundary>
          <Suspense fallback={<LoadingFallback />}>
            {renderPage()}
          </Suspense>
        </ErrorBoundary>
 
        {/* Notification Toast Container */}
        <ToastContainer />
      </main>
 
      {/* Spotlight Search (Ctrl+K) */}
      <SpotlightSearch
        isOpen={spotlightOpen}
        onClose={() => setSpotlightOpen(false)}
        onNavigate={(p) => { setPage(p as Page); setSpotlightOpen(false); }}
      />
 
      {/* AI Chatbot FAB */}
      <AIChatbot onNavigate={(p) => setPage(p as Page)} />
    </div>
  );
}
 
// App - tema kurulumu + auth guard.
// useEffect kosulsuz calisir; erken return bundan SONRA.
export default function App() {
  const authUser = useAuthStore((s) => s.user);
  const isAuthenticated = useAuthStore((s) => s.isAuthenticated);
  const logout = useAuthStore((s) => s.logout);
  const themeName = useUIStore((s) => s.themeName);
 
  useEffect(() => {
    syncRuntimeTheme(themeName);
  }, [themeName]);
 
  // Global 401 session-expired handler
  useEffect(() => {
    const handleSessionExpired = () => {
      logout();
    };
    window.addEventListener("auth:session-expired", handleSessionExpired);
    return () => window.removeEventListener("auth:session-expired", handleSessionExpired);
  }, [logout]);
 
  if (!isAuthenticated || !authUser) {
    return (
      <ToastProvider>
        <LoginPage />
      </ToastProvider>
    );
  }
 
  return (
    <ErrorBoundary>
      <ConfirmationProvider>
        <ToastProvider>
          <AuthenticatedApp authUser={authUser} />
        </ToastProvider>
      </ConfirmationProvider>
    </ErrorBoundary>
  );
}
